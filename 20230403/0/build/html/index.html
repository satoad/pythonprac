import calendar

year = calendar.year = 2023
month = calendar.month = 4

cal = calendar.monthcalendar(year, month)

table_header = "+---+---+---+---+---+---+---+\n|Mon|Tue|Wed|Thu|Fri|Sat|Sun|"

table_rows = ""
for week in cal:
    row = "|"
    for day in week:
        if day == 0:
            row += "   |"
        else:
            if day < 10:
                row += f" {day} |"
            else:
                row += f" {day}|"
    table_rows += f"+---+---+---+---+---+---+---+\n{row}\n"

table = f"{table_header}\n{table_rows}{table_header}\n+---+---+---+---+---+---+---+"

with open("calendar.rst", "w") as file:
    file.write(table)

print("Файл calendar.rst успешно создан!")
